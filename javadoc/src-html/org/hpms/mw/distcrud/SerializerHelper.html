001package org.hpms.mw.distcrud;
002
003import java.nio.ByteBuffer;
004import java.time.LocalDate;
005
006import javafx.scene.paint.Color;
007
008public class SerializerHelper {
009
010   public static void putString( String s, ByteBuffer target ) {
011      final byte[] bytes = s.getBytes();
012      target.putInt( bytes.length );
013      target.put( bytes );
014   }
015
016   public static String getString( ByteBuffer source ) {
017      final int len = source.getInt();
018      final byte[] bytes = new byte[len];
019      source.get( bytes );
020      return new String( bytes );
021   }
022
023   public static void putLocalDate( LocalDate date, ByteBuffer target ) {
024      target.putShort((short)date.getYear());
025      target.put     ((byte )date.getMonthValue());
026      target.put     ((byte )date.getDayOfMonth());
027   }
028
029   public static LocalDate getLocalDate( ByteBuffer source ) {
030      final short year       = source.getShort();
031      final byte  month      = source.get();
032      final byte  dayOfMonth = source.get();
033      return LocalDate.of( year, month, dayOfMonth );
034   }
035
036   public static void putFxColor( Color color, ByteBuffer target, Color def ) {
037      if( color == null ) {
038         color = def;
039      }
040      target.putDouble( color.getRed());
041      target.putDouble( color.getGreen());
042      target.putDouble( color.getBlue());
043      target.putDouble( color.getOpacity());
044   }
045
046   public static Color getFxColor( ByteBuffer source ) {
047      final double red     = source.getDouble();
048      final double green   = source.getDouble();
049      final double blue    = source.getDouble();
050      final double opacity = source.getDouble();
051      return Color.color( red, green, blue, opacity );
052   }
053}
