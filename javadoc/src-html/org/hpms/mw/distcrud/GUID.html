001package org.hpms.mw.distcrud;
002
003import java.nio.ByteBuffer;
004
005public final class GUID implements Comparable<GUID> {
006
007   static GUID unserialize( ByteBuffer buffer ) {
008      final GUID id = new GUID();
009      id._publisher = buffer.getShort();
010      id._cache     = buffer.get();
011      id._instance  = buffer.getInt();
012      assert id._publisher > 0;
013      assert id._cache     > 0;
014      assert id._instance  > 0;
015      return id;
016   }
017
018   short _publisher;
019   byte  _cache;
020   int   _instance;
021
022   boolean isShared() {
023      return _instance > 0;
024   }
025
026   void set( GUID id ) {
027      _publisher = id._publisher;
028      _cache     = id._cache;
029      _instance  = id._instance;
030      assert id._publisher > 0;
031      assert id._cache     > 0;
032      assert id._instance  > 0;
033   }
034
035   @Override
036   public String toString() {
037      return String.format( "Instance-%02X-%02X-%04X", _publisher, _cache, _instance );
038   }
039
040   @Override
041   public int compareTo( GUID right ) {
042      int delta = 0;
043      if( delta == 0 ) {
044         delta = this._publisher - right._publisher;
045      }
046      if( delta == 0 ) {
047         delta = this._cache     - right._cache;
048      }
049      if( delta == 0 ) {
050         delta = this._instance  - right._instance;
051      }
052      return delta;
053   }
054
055   public void serialize( ByteBuffer buffer ) {
056      buffer.putShort( _publisher );
057      buffer.put     ( _cache     );
058      buffer.putInt  ( _instance  );
059   }
060}
