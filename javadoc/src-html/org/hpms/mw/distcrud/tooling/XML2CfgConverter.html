001package org.hpms.mw.distcrud.tooling;
002
003import java.io.PrintStream;
004import java.util.LinkedHashMap;
005import java.util.Map;
006import java.util.Map.Entry;
007
008import javax.xml.parsers.DocumentBuilderFactory;
009
010import org.w3c.dom.Document;
011import org.w3c.dom.Element;
012import org.w3c.dom.NodeList;
013
014public final class XML2CfgConverter {
015
016   public static void main( String[] args ) throws Exception {
017      if( args.length != 2 ) {
018         System.err.println(
019            "Usage: " + XML2CfgConverter.class.getName() + " <xml file> <cfg file>" );
020         System.exit( 1 );
021      }
022      final String xml = args[0];
023      final String cfg = args[1];
024      final Document conf = DocumentBuilderFactory.newInstance().newDocumentBuilder().parse( xml );
025      final NodeList prtcpnts = conf.getElementsByTagName( "participant" );
026      final Map<String, Element> participants = new LinkedHashMap<>();
027      for( int i = 0, count = prtcpnts.getLength(); i < count; ++i ) {
028         final Element p        = (Element)prtcpnts.item( i );
029         final String  n        = p.getAttribute( "name" );
030         participants.put( n, p );
031      }
032      try( final PrintStream cfgStream = new PrintStream( cfg )) {
033         cfgStream.println( "#------- GENERATED FILE, DO NOT EDIT -------" );
034         cfgStream.println( "#-------------------------------------------" );
035         cfgStream.printf( "#%3s%18s%6s  %s\n", "id", "address", "port", "subscribe-to" );
036         cfgStream.println( "#-------------------------------------------" );
037         for( final Entry<String, Element> e : participants.entrySet()) {
038            final Element  elt         = e.getValue();
039            final String   id          = elt.getAttribute( "id" );
040            final String   address     = elt.getAttribute( "address" );
041            final String   port        = elt.getAttribute( "port" );
042            final NodeList subscribers = elt.getElementsByTagName( "subscribe" );
043            cfgStream.printf( " %3s%18s%6s  ", id, address, port );
044            for( int i = 0, count = subscribers.getLength(); i < count; ++i ) {
045               if( i > 0 ) {
046                  cfgStream.print( ',' );
047               }
048               final Element xSubscriber  = (Element)subscribers.item(i);
049               final String  subscriber   = xSubscriber.getAttribute( "to" );
050               final Element xParticipant = participants.get( subscriber );
051               assert xParticipant != null : subscriber + " not found!";
052               final String  refId        = xParticipant.getAttribute( "id" );
053               cfgStream.print( refId );
054            }
055            cfgStream.println();
056         }
057         cfgStream.println( "#-------------------------------------------" );
058         cfgStream.println( "#------- GENERATED FILE, DO NOT EDIT -------" );
059      }
060      System.out.printf( "%s successfully converted to %s", xml, cfg );
061   }
062}
